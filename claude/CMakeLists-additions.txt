# 4) Mac 应用名/图标：在 set_target_properties 那里改
if(APPLE)
    set_target_properties(${PROJECT_NAME} PROPERTIES
        MACOSX_BUNDLE TRUE
        MACOSX_BUNDLE_BUNDLE_NAME           "Cadence"
        MACOSX_BUNDLE_GUI_IDENTIFIER        "app.cadence.desktop"
        MACOSX_BUNDLE_BUNDLE_VERSION        "0.2.0"
        MACOSX_BUNDLE_SHORT_VERSION_STRING  "0.2.0"
        MACOSX_BUNDLE_ICON_FILE             "icon.icns"
    )

    # 把图标放进 .app/Contents/Resources/
    set(APP_ICON "${CMAKE_SOURCE_DIR}/build/icon.icns")
    if(EXISTS "${APP_ICON}")
        target_sources(${PROJECT_NAME} PRIVATE "${APP_ICON}")
        set_source_files_properties("${APP_ICON}" PROPERTIES
            MACOSX_PACKAGE_LOCATION "Resources")
    else()
        message(WARNING
            "icon.icns 还没生成。先在项目根目录跑 ./scripts/make_icns.sh，"
            "然后重新 cmake。")
    endif()
endif()
